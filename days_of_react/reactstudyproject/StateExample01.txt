# State

1. 컴포넌트의 기본적인 동작을 바꿀 때 쓰는 것은 props

2. 컴포넌트 내부적으로 사용되는 것들이 'State'라 부름

class App extends Component {
  constructor(props) {
    super(props);

    // State 사용
    this.state = { 
      subject : {title : "Web", sub:"World Wide Web!"}
      }
  }
  
  render() {
    return (
        <div className="App">
          <Subject 
          title={this.state.subject.title}
          sub={this.state.subject.sub}></Subject>
          <Toc></Toc
        </div>
    );
  }
}
